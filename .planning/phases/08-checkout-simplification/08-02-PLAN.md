  <name>Task 2: Simplify checkout machine — remove shipping/payment states</name>
  <files>web/src/machines/checkoutMachine.ts</files>
  <action>
Rewrite the checkout machine to have only 3 states: `contactInfo` → `submitting` → `success`.

Remove:
- `shipping` and `payment` states
- `SET_SHIPPING_METHOD` and `SET_PAYMENT_METHOD` events
- `shippingMethod` and `paymentMethod` from context
- `BACK` event (single step, no going back)

New machine:
```typescript
export type CheckoutContext = {
  submitError: string | null
  stockErrors: StockError[]
}

export type CheckoutEvent =
  | { type: 'SUBMIT'; formData: CreateOrderPayload }

export const checkoutMachine = setup({
  types: {
    context: {} as CheckoutContext,
    events: {} as CheckoutEvent,
    input: {} as { items: CartItem[] },
  },
  actors: {
    submitOrder: {
      src: async (_, { formData }: { formData: CreateOrderPayload }) => {
        const response = await createOrder(formData)
        return response.data
      },
    },
  },
}).createMachine({
  id: 'checkout',
  initial: 'contactInfo',
  context: {
    submitError: null,
    stockErrors: [],
  },
  states: {
    contactInfo: {
      on: {
        SUBMIT: {
          target: 'submitting',
          actions: assign({
            submitError: () => null,
            stockErrors: () => [],
          }),
        },
      },
    },
    submitting: {
      invoke: {
        src: 'submitOrder',
        input: ({ event }) => {
          if (event.type !== 'SUBMIT') throw new Error('Unexpected event')
          return { formData: event.formData }
        },
        onDone: 'success',
        onError: {
          target: 'contactInfo',
          actions: assign({
            submitError: ({ event }) => {
              const error = event.error as any
              if (error?.status === 409 && error?.data?.details) {
                return 'Algunos productos no tienen suficiente stock.'
              }
              return 'Error al procesar el pedido. Por favor intenta nuevamente.'
            },
            stockErrors: ({ event }) => {
              const error = event.error as any
              if (error?.status === 409 && error?.data?.details) {
                return error.data.details as StockError[]
              }
              return []
            },
          }),
        },
      },
    },
    success: {
      type: 'final',
    },
  },
})
```

Keep `StockError` type — stock errors still relevant.
